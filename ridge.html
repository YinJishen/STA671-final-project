corr_lag <- function(x){
  lag <- c(rep(NA, x), data[1:(dim(data)[1]-x),]$PM2.5)
  tmp <- data.frame(now = data$PM2.5,
                    lag = lag) %>% na.omit()
  return(cor(tmp$lag, tmp$now))
}

season <- function(x){
  if(x <= 5 & x >= 3){return("")}
  else if(x <= 8 & x >= 6){return("")}
  else if(x <= 11 & x >= 9){return("")}
  else{return("")}
}

level <- function(x){
  if(x < 35){return(1)}
  else if(x < 75){return(2)}
  else if(x < 115){return(3)}
  else if(x < 150){return(4)}
  else if(x < 250){return(5)}
  else{return(6)}
}

data1$season = factor(sapply(data1$month, season), levels = c("", "", "", ""))
